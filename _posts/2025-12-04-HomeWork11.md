## 1. Wiener Process (Brownian Motion) Simulation

The Wiener process $\( W(t) \)$ satisfies:
$$
W_{k+1} = W_k + \sqrt{\Delta t}\, Z_k, \quad Z_k \sim \mathcal{N}(0,1)
$$

## 2. Poisson Counting Process Simulation

Events occur independently at rate $\( \lambda \)$, approximated by:
$$
X_i \sim \text{Bernoulli}\left(\frac{\lambda}{n}\right)
$$

<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

<script>
let bmChart = null;
let poisChart = null;

/* Brownian Motion */
function runBrownian() {
  const n = parseInt(document.getElementById("bmSteps").value);
  const T = parseFloat(document.getElementById("bmT").value);
  const dt = T / n;

  let W = 0;
  let t = [];
  let path = [];

  for (let i = 0; i <= n; i++) {
    t.push(i * dt);
    path.push(W);
    W += Math.sqrt(dt) * gaussianRandom();
  }

  if (bmChart) bmChart.destroy();

  bmChart = new Chart(document.getElementById("bmChart"), {
    type: "line",
    data: {
      labels: t,
      datasets: [{
        label: "Wiener Process W(t)",
        data: path,
        borderWidth: 2,
        pointRadius: 0
      }]
    },
    options: {
      scales: {
        x: { title: { display: true, text: "t" }},
        y: { title: { display: true, text: "W(t)" }}
      }
    }
  });
}

/* Poisson Process */
function runPoisson() {
  const lambda = parseFloat(document.getElementById("lambda").value);
  const n = parseInt(document.getElementById("poisN").value);
  const p = lambda / n;

  let count = 0;
  let t = [];
  let N = [];

  for (let i = 0; i <= n; i++) {
    if (Math.random() < p) count++;
    t.push(i / n);
    N.push(count);
  }

  if (poisChart) poisChart.destroy();

  poisChart = new Chart(document.getElementById("poisChart"), {
    type: "line",
    data: {
      labels: t,
      datasets: [{
        label: "Counting Process N(t)",
        data: N,
        stepped: true,
        borderWidth: 2,
        pointRadius: 0
      }]
    },
    options: {
      scales: {
        x: { title: { display: true, text: "t" }},
        y: { title: { display: true, text: "N(t)" }}
      }
    }
  });
}

/* Standard Normal Generator */
function gaussianRandom() {
  let u = 0, v = 0;
  while (u === 0) u = Math.random();
  while (v === 0) v = Math.random();
  return Math.sqrt(-2.0 * Math.log(u)) * Math.cos(2.0 * Math.PI * v);
}
</script>

## 3. Interpretation

- **Brownian motion** has continuous paths, Gaussian increments, and variance proportional to time.
- **Poisson process** has jump paths, exponentially distributed interarrival times, and mean/variance equal to $\( \lambda t \)$.
- Both arise as limits of discrete random processes and are cornerstones of stochastic modeling.
